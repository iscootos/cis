如上，是1到1000的素数表                
通过如下代码获得
```c
#include <stdio.h>

int main(int argc, char *argv[])
{
	int prime = 1;
	int i, j;
	int count = 1;
	int isPrime;
	printf("2\t");
	for (i = 3; i < 1000; i++) {
		isPrime = 1;
		for ( j = 2; j < i; j++) {
			if (!(i%j)) {
				isPrime = 0;
				break;
			}
		}
		if (isPrime) {
			count++;
			printf("%d", i);
			if (count % 8)
				printf("\t");
			else
				printf("\n");
		}

	}

	return 0;
}
```
